turtles-own [
  hours ; How long it would take
  speed ; How fast would be the disease spread
]

globals [
  start-speed ; How fast would be the disease spread, we defiend that as globel that we could use it in the rest of the code
  social-distance ; One of the factors that would be used calculating the spread speed
  people-count ; The population size
  infection-probability ; How likely are people to contract the disease from contact
  incubation-time ; How long before first symptoms show up
  illness-duration ; How long are people ill
  severity-rate ; How many people would need to go to the hospital
  death-rate ; How many people do not survive in %
]


to setup
  clear-all
  reset-ticks
  set start-speed 0.35
  set people-count 600

 ; creat the turtles depending the people count
  create-turtles people-count [
    setxy random-xcor random-ycor
    set shape "circle"
    set color white
    set speed start-speed
  ]

  set social-distance 1.5  ; when people keep 1.5m distance
  set infection-probability 5 ; 5% chance of infection
  set incubation-time 5  ; 5 days incubation time
  set illness-duration 10  ; 10 days till people recoverd
  set severity-rate 20 ; 20% of the people do need to go to the hospital
  set death-rate 3.8 ; 3.8% of the people get dead

  ; start with 3 infected people
  ask n-of 3 turtles [
    set color orange
  ]
end


to go
  ask turtles [
    forward speed * (1 - 1 / social-distance)
    right 45 - random 90
    set hours hours + 1
  ]
  ; choose the color of the persoon. It could be red, orange, violet, green or gray.
  ask turtles [
    if (color = red) or (color = orange) [
      ask turtles in-radius 1 [
        if random 100 < infection-probability  [
          if color = white [
            set color orange
            set hours 0
          ]
        ]
      ]
    ]

    if (color = orange) and (hours > (incubation-time * 24)) [
      set color red
      set hours 0
      set speed 0
    ]

    if (color = red) and (hours = (illness-duration * 24)) [
      ifelse random 100 < severity-rate [
        set color violet
        set hours 0
        set speed 0
      ]
      [
        set color green
        set hours 0
        set speed start-speed
      ]
    ]

    if (color = violet) and (hours = (illness-duration * 24)) [
      ifelse random 100 < death-rate [
        set color gray
        set hours 0
        set speed 0
      ]
      [
        set color green
        set hours 0
        set speed start-speed
      ]
    ]
  ]
  ; when there are no orange, red or violet people the program should stop.
  if done? [stop]
  tick
end


to-report done?
  ; Gives the results to plot the 'infections graph'
  if  (count turtles with [color = red] = 0) and 
      (count turtles with [color = orange] = 0) and
      (count turtles with [color = violet] = 0)
  [
    report true
  ]

  report false
end
